/**
     * 从html中获取String类型的html
     * @param name  需要获取的html名称，例如,a.html对应 a
     *              html文件默认存储在resources中
     * @return      String类型的html
     */
    public static String getHtml(String name){
        String html="";
        try{
            Scanner in=new Scanner(Path.of("src\\main\\resources\\resources\\"+name+".html"));
            while(in.hasNext()){
                html+=in.next();
            }
        }catch (Exception e){
            e.printStackTrace();
        }
        return html;
    }
